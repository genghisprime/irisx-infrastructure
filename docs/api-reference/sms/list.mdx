## Overview

Retrieve a paginated list of SMS messages with filtering by status, direction, and date range.

## Authentication

<ParamField header="X-API-Key" type="string">
  API key for server-to-server authentication
</ParamField>

<ParamField header="Authorization" type="string">
  Bearer token for user authentication
</ParamField>

## Query Parameters

<ParamField query="status" type="string">
  Filter by status: `queued`, `sending`, `sent`, `delivered`, `failed`, `undelivered`
</ParamField>

<ParamField query="direction" type="string">
  Filter by direction: `inbound` or `outbound`
</ParamField>

<ParamField query="to" type="string">
  Filter by recipient number
</ParamField>

<ParamField query="from" type="string">
  Filter by sender number
</ParamField>

<ParamField query="start_date" type="string">
  Filter messages after this date (ISO 8601)
</ParamField>

<ParamField query="end_date" type="string">
  Filter messages before this date (ISO 8601)
</ParamField>

<ParamField query="page" type="integer" default={1}>
  Page number
</ParamField>

<ParamField query="limit" type="integer" default={50}>
  Results per page (max: 100)
</ParamField>

## Response

<ResponseField name="messages" type="array">
  Array of SMS message objects
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination metadata
</ResponseField>

```javascript Node.js
const axios = require('axios');

async function listSMS(filters = {}) {
  const params = new URLSearchParams({
    page: filters.page || 1,
    limit: filters.limit || 50,
    ...filters
  });

  const response = await axios.get(
    `https://api.tazzi.com/v1/sms?${params}`,
    {
      headers: { 'X-API-Key': 'YOUR_API_KEY' }
    }
  );

  const { messages, pagination } = response.data;
  console.log(`Found ${pagination.total} messages`);

  return response.data;
}

// Examples
listSMS({ status: 'delivered' });
listSMS({ direction: 'outbound', page: 2 });
```

```python Python
import requests

def list_sms(filters=None):
    if filters is None:
        filters = {}

    params = {
        'page': filters.get('page', 1),
        'limit': filters.get('limit', 50),
        **{k: v for k, v in filters.items() if k not in ['page', 'limit']}
    }

    response = requests.get(
        'https://api.tazzi.com/v1/sms',
        params=params,
        headers={'X-API-Key': 'YOUR_API_KEY'}
    )

    if response.status_code == 200:
        data = response.json()
        print(f"Found {data['pagination']['total']} messages")
        return data

list_sms({'status': 'delivered'})
```

## Filtering Examples

### Get today's delivered messages

```javascript
const today = new Date().toISOString().split('T')[0];
listSMS({
  status: 'delivered',
  start_date: `${today}T00:00:00Z`,
  end_date: `${today}T23:59:59Z`
});
```

### Get failed messages

```javascript
listSMS({ status: 'failed' });
```

## Analytics

Calculate messaging statistics:

```javascript
async function calculateSMSStats(filters) {
  const { messages } = await listSMS({ ...filters, limit: 100 });

  const stats = {
    total: messages.length,
    delivered: messages.filter(m => m.status === 'delivered').length,
    failed: messages.filter(m => m.status === 'failed').length,
    totalCost: messages.reduce((sum, m) => sum + (m.cost || 0), 0),
    totalSegments: messages.reduce((sum, m) => sum + m.segments, 0)
  };

  return stats;
}
```
