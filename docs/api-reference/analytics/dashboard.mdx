## Overview

Retrieve aggregated statistics across all communication channels for a given date range. Perfect for building dashboards and overview reports.

## Authentication

<ParamField header="X-API-Key" type="string">
  API key for server-to-server authentication
</ParamField>

<ParamField header="Authorization" type="string">
  Bearer token for user authentication
</ParamField>

## Query Parameters

<ParamField query="start_date" type="string">
  Start date for analytics (ISO 8601 format). Defaults to 30 days ago.
</ParamField>

<ParamField query="end_date" type="string">
  End date for analytics (ISO 8601 format). Defaults to now.
</ParamField>

```javascript Node.js
const axios = require('axios');

async function getDashboardAnalytics(startDate, endDate) {
  const params = new URLSearchParams();

  if (startDate) params.append('start_date', startDate);
  if (endDate) params.append('end_date', endDate);

  const response = await axios.get(
    `https://api.tazzi.com/v1/analytics/dashboard?${params}`,
    {
      headers: { 'X-API-Key': 'YOUR_API_KEY' }
    }
  );

  const stats = response.data;

  console.log('Calls:', stats.calls.total);
  console.log('SMS:', stats.sms.total);
  console.log('Emails:', stats.emails.total);
  console.log('Total Cost: $', (stats.calls.total_cost + stats.sms.total_cost).toFixed(2));

  return stats;
}

const startOfMonth = new Date();
```

```python Python
import requests
from datetime import datetime, timedelta

def get_dashboard_analytics(start_date=None, end_date=None):
    params = {}

    if start_date:
        params['start_date'] = start_date
    if end_date:
        params['end_date'] = end_date

    response = requests.get(
        'https://api.tazzi.com/v1/analytics/dashboard',
        params=params,
        headers={'X-API-Key': 'YOUR_API_KEY'}
    )

    if response.status_code == 200:
        stats = response.json()

        print(f"Calls: {stats['calls']['total']}")
        print(f"SMS: {stats['sms']['total']}")
        print(f"Emails: {stats['emails']['total']}")
        print(f"Total Cost: ${stats['calls']['total_cost'] + stats['sms']['total_cost']:.2f}")

        return stats

# Get last 30 days
start = (datetime.now() - timedelta(days=30)).isoformat()
end = datetime.now().isoformat()
get_dashboard_analytics(start, end)
```

## Calculate Success Rates

```javascript
function calculateSuccessRates(stats) {
  const callSuccessRate = (stats.calls.completed / stats.calls.total * 100).toFixed(2);
  const smsDeliveryRate = (stats.sms.delivered / stats.sms.total * 100).toFixed(2);
  const emailOpenRate = (stats.emails.opened / stats.emails.delivered * 100).toFixed(2);
  const emailClickRate = (stats.emails.clicked / stats.emails.delivered * 100).toFixed(2);

  return {
    calls: `${callSuccessRate}%`,
    sms: `${smsDeliveryRate}%`,
    emailOpen: `${emailOpenRate}%`,
    emailClick: `${emailClickRate}%`
  };
}
```

## Dashboard Widget Example

```javascript
async function renderDashboard() {
  const stats = await getDashboardAnalytics();
  const rates = calculateSuccessRates(stats);

  return {
    calls: {
      total: stats.calls.total,
      completed: stats.calls.completed,
      successRate: rates.calls,
      avgDuration: (stats.calls.total_duration / stats.calls.total / 60).toFixed(1) + ' min',
      totalCost: '$' + stats.calls.total_cost.toFixed(2)
    },
    sms: {
      total: stats.sms.total,
      delivered: stats.sms.delivered,
      deliveryRate: rates.sms,
      totalCost: '$' + stats.sms.total_cost.toFixed(2)
    },
    emails: {
      total: stats.emails.total,
      delivered: stats.emails.delivered,
      openRate: rates.emailOpen,
      clickRate: rates.emailClick
    }
  };
}
```
