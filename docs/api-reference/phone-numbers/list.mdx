## Overview

Retrieve a list of all phone numbers purchased for your account.

## Authentication

<ParamField header="X-API-Key" type="string">
  API key for server-to-server authentication
</ParamField>

<ParamField header="Authorization" type="string">
  Bearer token for user authentication
</ParamField>

## Query Parameters

<ParamField query="status" type="string">
  Filter by status: `active`, `suspended`, or `released`
</ParamField>

<ParamField query="type" type="string">
  Filter by type: `local`, `toll-free`, or `mobile`
</ParamField>

<ParamField query="country" type="string">
  Filter by country code
</ParamField>

## Response

<ResponseField name="phone_numbers" type="array">
  Array of phone number objects
</ResponseField>

<RequestExample>

```bash cURL
curl https://api.useiris.com/v1/phone-numbers \
  -H "X-API-Key: YOUR_API_KEY"
```

```javascript Node.js
const axios = require('axios');

async function listPhoneNumbers(filters = {}) {
  const params = new URLSearchParams(filters);

  const response = await axios.get(
    `https://api.useiris.com/v1/phone-numbers?${params}`,
    {
      headers: { 'X-API-Key': 'YOUR_API_KEY' }
    }
  );

  console.log('Found', response.data.phone_numbers.length, 'numbers');
  return response.data.phone_numbers;
}

// Examples
listPhoneNumbers();
listPhoneNumbers({ status: 'active' });
listPhoneNumbers({ type: 'toll-free' });
```

```python Python
import requests

def list_phone_numbers(filters=None):
    if filters is None:
        filters = {}

    response = requests.get(
        'https://api.useiris.com/v1/phone-numbers',
        params=filters,
        headers={'X-API-Key': 'YOUR_API_KEY'}
    )

    if response.status_code == 200:
        numbers = response.json()['phone_numbers']
        print(f"Found {len(numbers)} numbers")
        return numbers

list_phone_numbers()
```

## Calculate Total Cost

```javascript
async function calculateMonthlyCost() {
  const numbers = await listPhoneNumbers({ status: 'active' });

  const totalCost = numbers.reduce((sum, num) => sum + num.monthly_cost, 0);
  console.log('Total monthly cost: $', totalCost.toFixed(2));

  return totalCost;
}
```
