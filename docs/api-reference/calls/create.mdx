## Overview

Create an outbound voice call with intelligent carrier routing and automatic failover. IRISX will select the best carrier based on destination, cost, and quality metrics.

## Authentication

<ParamField header="X-API-Key" type="string">
  API key for server-to-server authentication
</ParamField>

<ParamField header="Authorization" type="string">
  Bearer token for user authentication
</ParamField>

<Note>
  Use either API key OR bearer token, not both.
</Note>

## Request Body

<ParamField body="to" type="string" required>
  Destination phone number in E.164 format (e.g., `+15551234567`)
</ParamField>

<ParamField body="from" type="string" required>
  Caller ID from your purchased phone numbers
</ParamField>

<ParamField body="webhook_url" type="string">
  URL to receive call status updates (highly recommended)
</ParamField>

<ParamField body="record" type="boolean" default={false}>
  Whether to record the call. Recording URL will be available after call completes.
</ParamField>

<ParamField body="timeout" type="integer" default={60}>
  Ring timeout in seconds (min: 10, max: 300)
</ParamField>

<ParamField body="metadata" type="object">
  Custom metadata to attach to the call (key-value pairs)
</ParamField>

const axios = require('axios');

async function makeCall() {
  try {
    const response = await axios.post(
      'https://api.useiris.com/v1/calls',
      {
        to: '+15551234567',
        from: '+15559876543',
        webhook_url: 'https://yourapp.com/webhooks/calls',
        record: true,
        timeout: 60,
        metadata: {
          customer_id: '12345',
          campaign: 'summer_sale'
        }
      },
      {
        headers: {
          'X-API-Key': 'YOUR_API_KEY',
          'Content-Type': 'application/json'
        }
      }
    );

    const call = response.data.call;
    console.log('Call initiated:', call.uuid);
    console.log('Status:', call.status);

    return call;

  } catch (error) {
    console.error('Call failed:', error.response?.data);
    throw error;
  }
}

## Call Status Lifecycle

```
initiated → ringing → in-progress → completed
                  ↓          ↓
                busy    no-answer → failed
```

<Tip>
  Use webhooks to receive real-time status updates. Polling the API is less efficient and may miss intermediate states.
</Tip>

## Webhook Payload

When you provide a `webhook_url`, IRISX will send POST requests for each status change:

```json
{
  "event": "call.completed",
  "timestamp": "2025-10-30T18:02:15Z",
  "data": {
    "uuid": "123e4567-e89b-12d3-a456-426614174000",

```

## Rate Limits

- **10 calls per minute** per account
- Burst allowed: up to 20 calls in 10 seconds
- Headers: `X-RateLimit-Limit`, `X-RateLimit-Remaining`, `X-RateLimit-Reset`

<Warning>
  If you exceed rate limits, you'll receive a `429 Too Many Requests` response. Wait until the reset time.
</Warning>

## Error Codes

| Code | Error | Description |
|------|-------|-------------|
| 400 | VALIDATION_ERROR | Invalid request parameters |
| 400 | INVALID_PHONE_NUMBER | Phone number format error |
| 401 | UNAUTHORIZED | Invalid API key or token |
| 403 | INSUFFICIENT_BALANCE | Not enough credits |
| 403 | NUMBER_NOT_OWNED | `from` number not in your account |
| 429 | RATE_LIMITED | Too many requests |
| 503 | NO_CARRIERS_AVAILABLE | All carriers unavailable (rare) |

## Best Practices

### Error Handling

```javascript
async function makeCallWithRetry(to, from, maxRetries = 3) {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await makeCall(to, from);
    } catch (error) {
      if (error.response?.status === 429) {
        // Rate limited - wait and retry
        const resetTime = error.response.headers['x-ratelimit-reset'];
        await sleep(resetTime - Date.now());
      } else if (error.response?.status >= 500) {
        // Server error - retry with exponential backoff
        await sleep(Math.pow(2, i) * 1000);
      } else {
        // Client error - don't retry
        throw error;
      }
    }
  }
  throw new Error('Max retries exceeded');
}
```

### Batch Calling

```javascript
async function makeCallBatch(recipients) {
  const results = [];

  for (const recipient of recipients) {
    try {
      const call = await makeCall(recipient.phone, '+15559876543');
      results.push({ phone: recipient.phone, success: true, uuid: call.uuid });

      // Respect rate limits - wait 6 seconds between calls
      await sleep(6000);

    } catch (error) {
      results.push({ phone: recipient.phone, success: false, error: error.message });
    }
  }

  return results;
}
```
