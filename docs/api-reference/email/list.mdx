## Overview

Retrieve a paginated list of emails with filtering by status and recipient.

## Authentication

<ParamField header="X-API-Key" type="string">
  API key for server-to-server authentication
</ParamField>

<ParamField header="Authorization" type="string">
  Bearer token for user authentication
</ParamField>

## Query Parameters

<ParamField query="status" type="string">
  Filter by status: `queued`, `sending`, `sent`, `delivered`, `opened`, `clicked`, `bounced`, `complained`, `failed`
</ParamField>

<ParamField query="to" type="string">
  Filter by recipient email
</ParamField>

<ParamField query="page" type="integer" default={1}>
  Page number
</ParamField>

<ParamField query="limit" type="integer" default={50}>
  Results per page (max: 100)
</ParamField>

## Response

<ResponseField name="emails" type="array">
  Array of email objects
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination metadata
</ResponseField>

```javascript Node.js
const axios = require('axios');

async function listEmails(filters = {}) {
  const params = new URLSearchParams({
    page: filters.page || 1,
    limit: filters.limit || 50,
    ...filters
  });

  const response = await axios.get(
    `https://api.useiris.com/v1/email?${params}`,
    {
      headers: { 'X-API-Key': 'YOUR_API_KEY' }
    }
  );

  return response.data;
}
```

```python Python
import requests

def list_emails(filters=None):
    if filters is None:
        filters = {}

    params = {
        'page': filters.get('page', 1),
        'limit': filters.get('limit', 50),
        **{k: v for k, v in filters.items() if k not in ['page', 'limit']}
    }

    response = requests.get(
        'https://api.useiris.com/v1/email',
        params=params,
        headers={'X-API-Key': 'YOUR_API_KEY'}
    )

    if response.status_code == 200:
        return response.json()
```

## Email Analytics

Calculate engagement metrics:

```javascript
async function calculateEmailStats() {
  const { emails } = await listEmails({ limit: 100 });

  const stats = {
    total: emails.length,
    delivered: emails.filter(e => e.status === 'delivered').length,
    opened: emails.filter(e => e.opened_at).length,
    clicked: emails.filter(e => e.clicked_at).length,
    bounced: emails.filter(e => e.status === 'bounced').length
  };

  stats.openRate = (stats.opened / stats.delivered * 100).toFixed(2);
  stats.clickRate = (stats.clicked / stats.delivered * 100).toFixed(2);

  return stats;
}
```
