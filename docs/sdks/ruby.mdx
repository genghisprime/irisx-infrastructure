## Installation

Install the IRISX SDK via Gemfile:

```ruby
gem 'irisx-sdk'
```

Or install directly:

```bash
gem install irisx-sdk
```

## Quick Start

```ruby
require 'irisx'

# Initialize with API key
irisx = Irisx::Client.new(api_key: 'your_api_key_here')

# Make your first call
call = irisx.calls.create(
  to: '+15551234567',
  from: '+15559876543',
  record: true
)

puts "Call initiated: #{call.uuid}"
```

## Authentication

### API Key (Recommended)

```ruby
require 'irisx'

irisx = Irisx::Client.new(api_key: 'your_api_key_here')
```

### JWT Token (For User Auth)

```ruby
require 'irisx'

irisx = Irisx::Client.new(token: 'your_jwt_token_here')

# Handle token refresh
irisx.on_token_refresh do |new_token|
  puts "Token refreshed: #{new_token}"
  # Save new token to secure storage
end
```

## Making Calls

### Create Outbound Call

```ruby
call = irisx.calls.create(
  to: '+15551234567',
  from: '+15559876543',
  webhook_url: 'https://yourapp.com/webhooks',
  record: true,
  timeout: 60,
  metadata: {
    customer_id: 'cust_123',
    order_id: 'order_456'
  }
)

puts "Call UUID: #{call.uuid}"
```

### Get Call Details

```ruby
call = irisx.calls.get('123e4567-e89b-12d3-a456-426614174000')

puts "Status: #{call.status}"
puts "Duration: #{call.duration}s"
puts "Cost: $#{call.cost}"
```

### List Calls

```ruby
calls = irisx.calls.list(
  status: 'completed',
  direction: 'outbound',
  start_date: '2025-10-01',
  end_date: '2025-10-31',
  page: 1,
  limit: 50
)

puts "Found #{calls.pagination.total} calls"

calls.data.each do |call|
  puts "#{call.uuid}: #{call.status}"
end
```

```ruby

  to: '+15551234567',
  from: '+15559876543',
  body: 'Check out this image!',
  media_urls: [
    'https://example.com/image.jpg'
  ]

```

### Bulk SMS

```ruby

  from: '+15559876543',
  recipients: [
    { to: '+15551111111', body: 'Message 1' },
    { to: '+15552222222', body: 'Message 2' },
    { to: '+15553333333', body: 'Message 3' }
  ]

puts "Sent: #{result.successful}, Failed: #{result.failed}"
```

### Using Templates


## Webhooks

### Create Webhook

```ruby

```

### Verify Webhook Signature

```ruby
require 'openssl'

def verify_webhook_signature(payload, signature, secret)
  expected_signature = OpenSSL::HMAC.hexdigest(
    'SHA256',
    secret,
    payload.to_json
  )

  OpenSSL::HMAC.secure_compare(expected_signature, signature)
end

# With Rails
class WebhooksController < ApplicationController
  skip_verification_before_action :verify_authenticity_token, only: :irisx

  def irisx
    signature = request.headers['X-IRISX-Signature']
    payload = request.body.read

    unless verify_webhook_signature(JSON.parse(payload), signature, ENV['IRISX_WEBHOOK_SECRET'])
      head :unauthorized
      return
    end

    # Process webhook
    event_data = JSON.parse(payload)
    event = event_data['event']
    data = event_data['data']

    head :ok
  end

  private

  def verify_webhook_signature(payload, signature, secret)
    expected_signature = OpenSSL::HMAC.hexdigest(
      'SHA256',
      secret,
      payload.to_json
    )

    OpenSSL::HMAC.secure_compare(expected_signature, signature)
  end
end
```

## Error Handling

```ruby
require 'irisx'
require 'irisx/errors'

begin
  call = irisx.calls.create(
    to: '+15551234567',
    from: '+15559876543'
  )
rescue Irisx::RateLimitError => e
  puts "Rate limited - retry after: #{e.retry_after}s"
  # Wait and retry
  sleep(e.retry_after)
rescue Irisx::ValidationError => e
  puts "Validation error: #{e.message}"
  puts "Details: #{e.details.inspect}"
rescue Irisx::Error => e
  puts "API Error: #{e.message}"
  puts "Code: #{e.code}"

  if e.code == 'INSUFFICIENT_FUNDS'
    puts "Low balance - add credits"
  end
rescue StandardError => e
  puts "Unexpected error: #{e.message}"
end
```

### Exception Types

Ruby SDK includes specific exception classes:

```ruby
require 'irisx/errors'

begin
  number = irisx.phone_numbers.get('invalid_id')
rescue Irisx::NotFoundError
  puts "Phone number not found"
rescue Irisx::AuthenticationError
  puts "Authentication failed - check your API key"
rescue Irisx::Error => e
  puts "Error: #{e.message}"
end
```

## Advanced Features

### Automatic Retry with Exponential Backoff

```ruby
irisx = Irisx::Client.new(
  api_key: 'your_api_key_here',
  retry: {
    enabled: true,
    max_retries: 3,
    base_delay: 1,      # 1 second
    max_delay: 30       # 30 seconds
  }
)
```

### Custom Timeout

```ruby
irisx = Irisx::Client.new(
  api_key: 'your_api_key_here',
  timeout: 60  # 60 seconds
)
```

### Logging

```ruby
require 'logger'

logger = Logger.new(STDOUT)
logger.level = Logger::DEBUG

irisx = Irisx::Client.new(
  api_key: 'your_api_key_here',
  logger: logger
)
```
