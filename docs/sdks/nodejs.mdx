## Installation

Install the Tazzi SDK via npm:

```bash
npm install @tazzi/sdk
```

Or with yarn:

```bash
yarn add @tazzi/sdk
```

## Quick Start

```javascript
const Tazzi = require('@tazzi/sdk');

// Initialize with API key
const tazzi = new Tazzi({
  apiKey: process.env.TAZZI_API_KEY
});

// Make your first call
async function makeCall() {
  const call = await tazzi.calls.create({
    to: '+15551234567',
    from: '+15559876543',
    record: true
  });

  console.log(`Call initiated: ${call.uuid}`);
}

makeCall();
```

```javascript

  token: 'your_jwt_token_here'
});

// Auto-refresh token
tazzi.onTokenRefresh((newToken) => {
  console.log('Token refreshed:', newToken);
  // Save new token
});
```

## Making Calls

### Create Outbound Call

```javascript
const call = await tazzi.calls.create({
  to: '+15551234567',
  from: '+15559876543',
  webhook_url: 'https://yourapp.com/webhooks',
  record: true,
  timeout: 60,
  metadata: {
    customer_id: 'cust_123',
    order_id: 'order_456'
  }
});
```

### Get Call Details

```javascript

console.log(`Status: ${call.status}`);
console.log(`Duration: ${call.duration}s`);
console.log(`Cost: $${call.cost}`);
```

### List Calls

```javascript
const calls = await tazzi.calls.list({
  status: 'completed',
  direction: 'outbound',
  start_date: '2025-10-01',
  end_date: '2025-10-31',
  page: 1,
  limit: 50
});

console.log(`Found ${calls.pagination.total} calls`);

calls.data.forEach(call => {
  console.log(`${call.uuid}: ${call.status}`);
});
```

## Webhooks

### Create Webhook

```javascript

  url: 'https://yourapp.com/webhooks/tazzi',

    'call.failed',
    'sms.delivered',
    'sms.failed'
  ],
  description: 'Production webhook'

```

### List Webhooks

```javascript
const webhooks = await tazzi.webhooks.list();

webhooks.forEach(webhook => {
  console.log(`${webhook.id}: ${webhook.url}`);
});
```

### Verify Webhook Signature

```javascript
const crypto = require('crypto');

function verifyWebhook(payload, signature, secret) {
  const expectedSignature = crypto
    .createHmac('sha256', secret)
    .update(JSON.stringify(payload))
    .digest('hex');

  return signature === expectedSignature;
}

// In Express.js
app.post('/webhooks/tazzi', (req, res) => {
  const signature = req.headers['x-tazzi-signature'];
  const isValid = verifyWebhook(req.body, signature, webhookSecret);

  if (!isValid) {
    return res.status(401).send('Invalid signature');
  }

  // Process webhook
  const { event, data } = req.body;
  // ...

  res.status(200).send('OK');
});
```

## Error Handling

```javascript
try {
  const call = await tazzi.calls.create({ /* ... */ });
} catch (error) {
  if (error.code === 'RATE_LIMITED') {
    console.log('Rate limited - retry after:', error.retry_after);
  } else if (error.code === 'INVALID_NUMBER') {
    console.log('Invalid phone number format');
  } else if (error.code === 'INSUFFICIENT_FUNDS') {
    console.log('Low balance - add credits');
  } else {
    console.error('API Error:', error.message);
  }
}
```

### Error Types

All SDK errors extend `TazziError`:

```javascript
const { TazziError, RateLimitError, ValidationError } = require('@tazzi/sdk');

try {
  // ...
} catch (error) {
  if (error instanceof RateLimitError) {
    // Wait and retry
    await sleep(error.retry_after * 1000);
  } else if (error instanceof ValidationError) {
    // Fix request parameters
    console.log('Validation errors:', error.details);
  } else if (error instanceof TazziError) {
    // Other API error
    console.log('API Error:', error.message);
  } else {
    // Network or other error
    throw error;
  }
}
```

## Advanced Features

### Automatic Retry with Exponential Backoff

```javascript
const tazzi = new Tazzi({
  apiKey: process.env.TAZZI_API_KEY,
  retry: {
    enabled: true,
    max_retries: 3,
    base_delay: 1000, // 1 second
    max_delay: 30000  // 30 seconds
  }
});
```

### Custom Timeout

```javascript
const tazzi = new Tazzi({
  apiKey: process.env.TAZZI_API_KEY,
  timeout: 60000 // 60 seconds
});
```

### Logging

```javascript
const tazzi = new Tazzi({
  apiKey: process.env.TAZZI_API_KEY,
  logger: console, // or custom logger
  logLevel: 'debug' // 'error', 'warn', 'info', 'debug'
});
```

## TypeScript Support

The SDK is written in TypeScript with full type definitions:

```typescript
import Tazzi, { Call, SMS, Email } from '@tazzi/sdk';

const tazzi = new Tazzi({

});

const call: Call = await tazzi.calls.create({

  record: true
});
```
